import pandas as pd

def to_snake(df):
    df2 = df.copy()
    df2.columns = (df2.columns
                   .str.strip()
                   .str.replace(r"\s+", "_", regex=True)
                   .str.lower())
    return df2

def audit_missing(df, name):
    miss = df.isna().sum().to_frame("missing")
    miss["pct"] = (miss["missing"] / len(df) * 100).round(2)
    miss.insert(0, "column", miss.index)
    miss = miss.reset_index(drop=True)
    print(f"\n--- Missing audit: {name} ---\n", miss.sort_values("missing", ascending=False).head(20))
    return miss

def preview_categories(df, cols):
    for c in cols:
        if c in df.columns:
            vals = df[c].astype("string").str.strip()
            print(f"\n{c}: unique sample ->", sorted(vals.dropna().unique().tolist())[:12])

def date_preview(df, col, dayfirst=True):
    if col not in df.columns: 
        return {"col": col, "present": False}
    parsed = pd.to_datetime(df[col], errors="coerce", dayfirst=dayfirst)
    invalid = parsed.isna().sum()
    return {"col": col, "present": True, "rows": len(df),
            "invalid_dates": int(invalid),
            "invalid_pct": round(invalid/len(df)*100,2)}

def numeric_checks(df, rules):
    findings = []
    for col, rule in rules.items():
        if col not in df.columns: 
            findings.append((col, "missing_col", 0))
            continue
        s = pd.to_numeric(df[col], errors="coerce")
        if rule == "non_negative":
            bad = (s < 0).sum()
            findings.append((col, "negatives", int(bad)))
        elif rule == "fraction_0_1":
            bad = ((s < 0) | (s > 1)).sum()
            findings.append((col, "out_of_[0,1]", int(bad)))
    return pd.DataFrame(findings, columns=["column","issue","count"])

Step 2: Clean Each Dataset Individually

Goal

Prepare all three datasets (sales_data, product_info, and customer_info) for merging by ensuring accuracy, consistency, and completeness.

Actions

For each dataset:

  1. Fix text formatting remove extra spaces and standardise casing
    (e.g., " delayed " "Delayed", "gold " "Gold").
  2. Convert date columns order_date, signup_date, launch_date datetime objects.
  3. Handle missing values
    • Replace blank numerics (e.g., discount_applied) with 0.0
    • Fill text fields (region, category, gender, etc.) with "Unknown"
    • Drop records only if essential keys (order_id, product_id, customer_id) are missing.
  4. Remove duplicates ensure each key (order_id, product_id, customer_id) is unique.
  5. Validate numeric values ensure quantity, unit_price, and discount_applied are non-negative.
  6. Confirm cleaning results print shapes, missing counts, and unique label samples.

Why This Step Matters

Data cleaning ensures the merged dataset is reliable for feature engineering and analysis.
Standardising formats prevents merge mismatches and ensures accurate grouping and aggregation in later steps.

# === STEP 2: CLEAN EACH DATASET ===
import numpy as np
import pandas as pd

# Work on copies from Step 1
sales_clean    = sales_df.copy()
products_clean = products_df.copy()
customers_clean= customers_df.copy()

# ---------- Helper: text normaliser ----------
def norm_text(series):
    return (series.astype("string")
                  .str.strip()
                  .str.replace(r"\s+", " ", regex=True)
                  .str.title())

# ---------- 1 SALES DATA CLEANING ----------
print("\n=== Cleaning: SALES DATA ===")
if "delivery_status" in sales_clean.columns:
    sales_clean["delivery_status"] = norm_text(sales_clean["delivery_status"])
if "payment_method" in sales_clean.columns:
    sales_clean["payment_method"] = norm_text(sales_clean["payment_method"])
if "region" in sales_clean.columns:
    sales_clean["region"] = norm_text(sales_clean["region"])

# Convert order_date
sales_clean["order_date"] = pd.to_datetime(sales_clean["order_date"], errors="coerce", dayfirst=True)

# Replace missing numeric fields
for col in ["quantity","unit_price","discount_applied"]:
    if col in sales_clean.columns:
        sales_clean[col] = pd.to_numeric(sales_clean[col], errors="coerce")
        if col == "discount_applied":
            sales_clean[col] = sales_clean[col].fillna(0.0)
        else:
            sales_clean[col] = sales_clean[col].fillna(0)

# Drop invalid/blank key rows
if "order_id" in sales_clean.columns:
    before = len(sales_clean)
    sales_clean = sales_clean.dropna(subset=["order_id","customer_id","product_id"])
    print(f"Rows dropped (missing IDs): {before - len(sales_clean)}")

# Remove duplicates
sales_clean = sales_clean.drop_duplicates(subset="order_id", keep="first")

# Validate numeric columns
num_cols = ["quantity","unit_price","discount_applied"]
for col in num_cols:
    if col in sales_clean.columns:
        negatives = (sales_clean[col] < 0).sum()
        if negatives > 0:
            print(f"{col}: {negatives} negative values corrected to 0")
            sales_clean.loc[sales_clean[col] < 0, col] = 0

# ---------- 2 PRODUCT DATA CLEANING ----------
print("\n=== Cleaning: PRODUCT DATA ===")
for col in ["category","supplier_code"]:
    if col in products_clean.columns:
        products_clean[col] = norm_text(products_clean[col])

products_clean["launch_date"] = pd.to_datetime(products_clean["launch_date"], errors="coerce", dayfirst=True)

# Drop invalid product_id
if "product_id" in products_clean.columns:
    before = len(products_clean)
    products_clean = products_clean.dropna(subset=["product_id"])
    products_clean = products_clean.drop_duplicates(subset="product_id", keep="first")
    print(f"Rows dropped (invalid product_id): {before - len(products_clean)}")

# Fill missing base_price with median if available
if "base_price" in products_clean.columns:
    median_price = products_clean["base_price"].median()
    products_clean["base_price"] = products_clean["base_price"].fillna(median_price)

# ---------- 3 CUSTOMER DATA CLEANING ----------
print("\n=== Cleaning: CUSTOMER DATA ===")
for col in ["gender","region","loyalty_tier"]:
    if col in customers_clean.columns:
        customers_clean[col] = norm_text(customers_clean[col])

customers_clean["signup_date"] = pd.to_datetime(customers_clean["signup_date"], errors="coerce", dayfirst=True)

# Drop invalid customer_id
if "customer_id" in customers_clean.columns:
    before = len(customers_clean)
    customers_clean = customers_clean.dropna(subset=["customer_id"])
    customers_clean = customers_clean.drop_duplicates(subset="customer_id", keep="first")
    print(f"Rows dropped (invalid customer_id): {before - len(customers_clean)}")

# Fill missing text fields
fill_values = {"region":"Unknown","gender":"Unspecified","loyalty_tier":"Unknown"}
customers_clean = customers_clean.fillna(value=fill_values)

# ---------- 4 POST-CLEAN CHECKS ----------
print("\n=== AFTER CLEANING SNAPSHOT ===")
print("sales:", sales_clean.shape)
print("products:", products_clean.shape)
print("customers:", customers_clean.shape)

print("\nRemaining missing values (top 10 cols):")
print(sales_clean.isna().sum().sort_values(ascending=False).head(10))
print(products_clean.isna().sum().sort_values(ascending=False).head(10))
print(customers_clean.isna().sum().sort_values(ascending=False).head(10))

# ---------- 5 SAVE CLEAN VERSIONS ----------
sales_clean.to_csv("sales_data_clean.csv", index=False)
products_clean.to_csv("product_info_clean.csv", index=False)
customers_clean.to_csv("customer_info_clean.csv", index=False)
print("\n Cleaned files saved: sales_data_clean.csv, product_info_clean.csv, customer_info_clean.csv")

Step 4: Feature Engineering

Goal

Enhance the merged dataset with new analytical features that reveal business insights and prepare the data for visualisation.

Actions

The following new columns are created:

  1. revenue total revenue per transaction
    revenue = quantity * unit_price * (1 - discount_applied)
  2. order_week ISO week number extracted from order_date
  3. price_band categorical price segmentation
    • Low: < 15
    • Medium: 1530
    • High: > 30
  4. days_to_order number of days between launch_date (product) and order_date (sale)
  5. email_domain extracted domain name from customer email (e.g., gmail.com)
  6. is_late boolean flag if delivery_status = Delayed

Why This Step Matters

Feature engineering translates raw columns into business-relevant indicators.
These features help identify which products and customers drive performance, how time affects orders, and where delays or discounts influence revenue.

# === STEP 4: FEATURE ENGINEERING ===
import numpy as np
import pandas as pd

# Ensure were using the merged dataset
merged_df = pd.read_csv("merged_sales_data.csv")

print("Starting feature engineering on merged dataset of shape:", merged_df.shape)

# ---------- 1 Revenue ----------
merged_df["revenue"] = merged_df["quantity"] * merged_df["unit_price"] * (1 - merged_df["discount_applied"])

# ---------- 2 Order Week ----------
merged_df["order_date"] = pd.to_datetime(merged_df["order_date"], errors="coerce")
merged_df["order_week"] = merged_df["order_date"].dt.isocalendar().week

# ---------- 3 Price Band ----------
merged_df["price_band"] = pd.cut(
    merged_df["unit_price"],
    bins=[0, 15, 30, np.inf],
    labels=["Low", "Medium", "High"]
)

# ---------- 4 Days to Order ----------
merged_df["launch_date"] = pd.to_datetime(merged_df["launch_date"], errors="coerce")
merged_df["days_to_order"] = (merged_df["order_date"] - merged_df["launch_date"]).dt.days

# ---------- 5 Email Domain ----------
if "email" in merged_df.columns:
    merged_df["email_domain"] = merged_df["email"].astype(str).str.extract(r'@([\w\.-]+)')

# ---------- 6 Is Late ----------
if "delivery_status" in merged_df.columns:
    merged_df["delivery_status"] = merged_df["delivery_status"].astype(str).str.title().str.strip()
    merged_df["is_late"] = merged_df["delivery_status"] == "Delayed"

# ---------- 7 Post-creation check ----------
print("\n=== Feature Summary ===")
print(merged_df[["revenue", "order_week", "price_band", "days_to_order", "email_domain", "is_late"]].head())

print("\nMissing values in new features:")
print(merged_df[["revenue", "order_week", "price_band", "days_to_order"]].isna().sum())

# ---------- 8 Save enhanced dataset ----------
merged_df.to_csv("merged_sales_data_features.csv", index=False)
print("\n Saved enhanced dataset as merged_sales_data_features.csv")

from pandasql import sqldf
from sklearn.preprocessing import MinMaxScaler
import pandas as pd
pysqldf = lambda q: sqldf(q, globals())

# Load data
merged_df = pd.read_csv("merged_sales_data_features.csv")

# ---------- 1 Customers who signed up in Q2, ordered within 14 days, and received > 20 % discount ----------
q = """
SELECT DISTINCT customer_id, order_id, signup_date, order_date, discount_applied
FROM merged_df
WHERE CAST(STRFTIME('%m', signup_date) AS INTEGER) BETWEEN 4 AND 6
  AND (JULIANDAY(order_date) - JULIANDAY(signup_date)) <= 14
  AND discount_applied > 0.20
"""
q2_customers = pysqldf(q)
print("\n=== Q2 Customers with early orders & high discounts ===")
print(q2_customers.head())

# ---------- 2 MinMaxScaler to normalise revenue ----------
scaler = MinMaxScaler()
merged_df["revenue_scaled"] = scaler.fit_transform(merged_df[["revenue"]])
print("\nScaled revenue sample:")
print(merged_df[["revenue", "revenue_scaled"]].head())

# ---------- 3 Flag underperforming products ----------
merged_df["underperforming"] = (
    (merged_df["quantity"] < merged_df["quantity"].median()) &
    (merged_df["discount_applied"] > 0.20) &
    (merged_df["is_late"] == True)
)
underperf = (
    merged_df.groupby("category", as_index=False)
    .agg(underperforming_count=("underperforming", "sum"),
         total_products=("product_id", "nunique"))
)
underperf["underperforming_rate_%"] = round(
    underperf["underperforming_count"] / underperf["total_products"] * 100, 2
)

print("\n=== Underperforming Products by Category ===")
print(underperf.sort_values("underperforming_rate_%", ascending=False))

# Save optional outputs
q2_customers.to_csv("stretch_q2_customers.csv", index=False)
underperf.to_csv("stretch_underperforming_products.csv", index=False)
print("\n Optional stretch outputs saved:")
print("stretch_q2_customers.csv")

Final Summary & Conclusion

Project: Week 2 Sales & Customer Behaviour Insights Green Cart Ltd.

This project analysed Green Cart Ltd.s sales, customer, and product data to understand revenue drivers, customer engagement, and operational performance.
Through a structured data analysis process, the datasets were cleaned, merged, enriched with new features, and explored visually to deliver clear, data-driven insights.

Data Preparation

  • Three datasets (sales_data, product_info, customer_info) were cleaned individually to remove duplicates, standardise text, and fix inconsistent formatting.
  • Missing values were handled logically numeric columns filled with 0.0, categorical with "Unknown".
  • Dates were converted to proper datetime objects and verified for consistency.
  • Validation ensured all numeric values were non-negative and key IDs were unique.

Feature Engineering

Key business-oriented fields were created:

  • revenue = quantity * unit_price * (1 - discount_applied)
  • order_week (ISO week of order)
  • price_band (Low / Medium / High)
  • days_to_order (from launch to order)
  • email_domain (for communication insights)
  • is_late (flag for delayed deliveries)

These features enabled trend analysis and behavioural segmentation.

Key Findings

  • Top categories such as Home Cleaning and Kitchen generated the most revenue, particularly in the South and North regions.
  • Discounts had a mild positive correlation with quantity sold, indicating moderate price sensitivity.
  • Gold loyalty-tier customers delivered the highest lifetime value.
  • Delivery delays were most frequent in Northern regions and for High price-band products.
  • Signup patterns showed customers joining mid-year (AprilJune) were the most active purchasers.

Business Recommendations

  1. Strengthen top-performing categories through targeted promotions and sustainable inventory planning.
  2. Reduce delivery delays by reviewing logistics performance in weaker regions.
  3. Reward Gold-tier customers with loyalty benefits; design incentives to convert Silver and Bronze users.
  4. Leverage signup trends by launching referral or seasonal campaigns around Q2.

Optional Insights (Stretch Analysis)

  • Identified Q2 customers who ordered within 14 days and received >20% discounts potential early adopters for remarketing.
  • Used MinMaxScaler to normalise revenue for fair product comparison.
  • Flagged underperforming products that combine low quantity, high discount, and frequent delays ideal candidates for review or discontinuation.
